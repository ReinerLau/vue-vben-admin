sequenceDiagram
  participant A as CountButton.vue
  participant B as useCountdown.ts

  B-->>-A: currentCount = 60

  B-->>-A: isStart = true
  loop setInterval 1000ms
    alt currentCount === 1
      activate B
      B--)A: isStart = false
      B--)A: currentCount = 60
      deactivate B

      activate B
      B--)A: currentCount - 1
      deactivate B
    end
  end
